import { clerkMiddleware, createRouteMatcher }
  from '@clerk/nextjs/server';
import { NextResponse } from 'next/server';
import { resolveUserAccess } from '@/lib/auth/rbac';

const isPublic = createRouteMatcher([
  '/', '/sign-in(.*)', '/sign-up(.*)'
]);

export default clerkMiddleware(async (auth, req) => {
  if (isPublic(req)) return NextResponse.next();
  const { userId, orgId, orgRole } =
    await auth.protect();
  const access = await resolveUserAccess(
    userId, orgId ?? ''
  );
  const headers = new Headers(req.headers);
  headers.set('x-current-org-id', orgId ?? '');
  headers.set('x-current-user-id',
    access.internal_user_id ?? '');
  headers.set('x-current-user-role',
    access.role ?? 'member');
  headers.set('x-current-user-dept-id',
    access.dept_id ?? '');
  headers.set('x-current-accessible-depts',
    JSON.stringify(access.accessible_dept_ids ?? []));
  return NextResponse.next(
    { request: { headers } }
  );
});
export const config = { matcher: [

]};
